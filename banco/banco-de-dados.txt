CREATE TABLE Medico (
	id SERIAL PRIMARY KEY NOT NULL,
	nome_medico varchar(50) NOT NULL,
	num_crm int NOT NULL,
	especialidade varchar(20) NOT NULL
)
	
CREATE TABLE Paciente (
	id SERIAL PRIMARY KEY NOT NULL,
	nome_paciente varchar(50) NOT NULL,
	cod_sus int NOT NULL,
	cpf varchar(11) NOT NULL
)
	
CREATE TABLE CentroCirurgico (
	id SERIAL PRIMARY KEY NOT NULL
)

CREATE TABLE Reserva (
	id SERIAL PRIMARY KEY,
	data_hora timestamp NOT NULL,
	fk_centrocirurgico int REFERENCES CentroCirurgico(id) NOT NULL
)

CREATE TABLE Cirurgia(
	id SERIAL PRIMARY KEY,
	fk_paciente int REFERENCES Paciente(id) NOT NULL,
	fk_medico int REFERENCES Medico(id) NOT NULL,
	fk_reserva int REFERENCES Reserva(id) NOT NULL
)
